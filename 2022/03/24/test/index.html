AC代码

#include <vector>
#include <iostream>
using namespace std;
int main()
{
    int n, N;
    cin >> n >> N;
    vector<int> A;
    A.emplace_back(0);
    for (int i = 1; i <= n; ++i)
    {
        int A_i;
        cin >> A_i;
        A.emplace_back(A_i);
    }
    A.emplace_back(N);
    int sum = 0;
    for (int i = 0; i <= n; ++i)
    {
        sum += (A[i + 1] - A[i]) * i;
    }
    cout << sum << endl;
    return 0;
}
